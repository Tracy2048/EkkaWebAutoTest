    const couponEle = $('.code-coupon');

    couponEle.on('click', function() {
        const $this = $(this);
        const couponCode = $this.text().trim();

        navigator.clipboard
            .writeText(couponCode)
            .then(function() {
                $this.text('Copied!');
            })
            .catch(function(err) {
                console.error('Không thể sao chép mã: ', err);
            });
    });
